#Solution for question 1


apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: nginx
  name: deploy-test
spec:
 replicas: 2
 selector:
   matchLabels:
     app: nginx
 template:
   metadata:
     labels:
       app: nginx
   spec:
     containers:
     - image: nginx
       name: container-test



kubectl set image deployment deploy-test container-test=nginx:alpine --record=true

kubectl rollout status deployment deploy-test
